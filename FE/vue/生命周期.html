    生命周期

    # 生命周期

    vue 实例从创建到销毁的过程就是 vue 的生命周期。指从创建，初始化数据，编译模版，挂载 dom 到渲染，更新到渲染，销毁等一系列过程。它主要分为 8 个阶段创建前后，载入前后，更新前后和销毁前后以及一些特殊场景的生命周期：比如 activated 生命周期是 keep-alive 组件缓存组件激活时调用。deactivated，keep-alive 缓存的组件停用时调用

    运行如下代码进行验证

    <template>
    <div class="home" ref="contianer">
        <p>{{ fullName }}</p>
        <button @click="handleClick"> change data </button>
    </div>
</template>

<script>
export default {
    name: 'LifeCycle',
    data(){
        return{
            fullName: 'naomi'
        }
    },
    // 创建阶段
    beforeCreate() {
        // 实例刚在内存中创建出来，还没有初始化 data 和 methods
        console.log('beforeCreate')
        console.log(`this.fullName:${this.fullName}`)
        console.log(`this.handleClick:${this.handleClick}`)
    },
    create() {
        // 实例已经内存中创建完成， 此时 data 和 methods 也已经创建完成
        console.log('create')
        console.log(`this.fullName:${this.fullName}`)
        console.log(`this.handleClick:${this.handleClick}`)
    },
    beforeMount() {
        // 此时已经完成了模版的编译，只是还没有渲染到页面中去
        console.log('beforeMount')
        console.log(`this.$refs:${this.$refs.contianer}`)
        
    },
    mounted() {
        // 模版已经渲染到了浏览器，创建阶段结束，即将进入运行阶段
        console.log('mount')
        console.log(`this.$refs:${this.$refs.contianer}`)
    },
    // 运行阶段
    beforeUpdate() {
        // 数据更新时调用，这里适合在更新之前访问 DOM
        console.log('beforeUpdate')
        console.log(`this.fullName:${this.fullName}`)

    },
    update() {
        // 页面重新渲染完毕，页面中数据与 data 一致
        console.log('update')
        console.log(`更新结束页面数据已更新`)
    },
    // 销毁阶段
    beforeDestroy() {
        // 在实例销毁之前调用，实例完全可用
        console.log('beforeDestroy')
    },
    destroy() {
        // 实例销毁之后调用，对应的 vue 实例所有指令都被解绑，所有的事件监听器被移除，所有的子实例也被销毁
        console.log('destroy')
    },
    methods: {
        handleClick() {
            this.fullName = 'acsn'
        }
    },
}
</script>

    # vue 父子组件生命周期调用顺序

    # 加载渲染

    父组件 beforecreated -> 父组件 created -> 父组件 beforeMount -> 子组件 beforeCreated -> 子组件 created -> 子组件 beforeMount -> 子组件 mount -> 父组件 mount

    # 子组件更新过程

    父组件 beforeUpdate -> 子组件 beforeUpdate -> 子组件 update -> 父组件 update

    # 父组件更新过程

    父组件 beforeUpdate -> 父组件 update

    # 销毁过程

    父组件 beforeDestroy -> 子组件 beforeDestroy -> 子组件 destroy -> 父组件 destroy
